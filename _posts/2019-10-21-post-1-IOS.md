## UI color
### <span style="color:#646767; margin-left: 1rem; line-height: 1.8rem; padding-top: 0.2rem;"> ∙ hex color function</span> 
- hex.swift로 만들어 씀. 쓰고싶은 View에서 extension 함

```swift

import Foundation
import UIKit

extension ViewController {
    
    func hexStringToUIColor (hex:String) -> UIColor {
        var cString:String = hex.trimmingCharacters(in: .whitespacesAndNewlines).uppercased()
        
        if (cString.hasPrefix("#")) {
            cString.remove(at: cString.startIndex)
        }
        
        if ((cString.count) != 6) {
            return UIColor.gray
        }
        
        var rgbValue:UInt32 = 0
        Scanner(string: cString).scanHexInt32(&rgbValue)
        
        return UIColor(
            red: CGFloat((rgbValue & 0xFF0000) >> 16) / 255.0,
            green: CGFloat((rgbValue & 0x00FF00) >> 8) / 255.0,
            blue: CGFloat(rgbValue & 0x0000FF) / 255.0,
            alpha: CGFloat(1.0)
        )
    }
}

```
사용은 요렇게 편함

func timeStart() {

    //카운트 값 초기화 하고 카운트 시작됨
    resetCount()

    //timeStop 타이머도 종료하고 카운트 재 시작
    scoreTimer.invalidate()

    scoreTimer = Timer.scheduledTimer(timeInterval: 1, target: self, selector: #selector(counter), userInfo: nil, repeats: true)

}

//일시정지
func timeStop() {
    scoreTimer.invalidate()
}

    print("resume")
    //timeStop 타이머 켤때 항상 타이머 끄고 시작할것,
    scoreTimer.invalidate()
    scoreTimer = Timer.scheduledTimer(timeInterval: 1, target: self, selector: #selector(counter), userInfo: nil, repeats: true)

}

//카운팅될 스코어 계산 로직
@objc func counter() {

    //누적 계산
    sec += 1
    if sec == 60 {
        min += 1
        sec = 0
        if min == 60 {
            hou += 1
            min = 0
        }
    }

    //계산 표시
    if sec < 10 {
        secString = "0\(sec)"
    }else {secString = "\(sec)"}

    if min < 10 {
        minString = "0\(min)"
    }else {minString = "\(min)"}

        houString = "0\(hou):"
